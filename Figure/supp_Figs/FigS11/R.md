data <- read.table("Finale_Figure/Figure1/1B_1C/Orthogroups_simulation_100.stats",header=T)

summ_data <- data %>%
   group_by(Number,Type) %>%
   summarise(qs = quantile(Gene, c(0.05, 0.95)),prob = c("p05", "p95"),mean=mean(Gene))
